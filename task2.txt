select
 (@index := @index + 1) as id,
 substring_index(name, ' ', -1) as type
FROM
 (select @index := 0) as t_index,

group by type;

 then substring(raw_names.name, 1, char_length(raw_names.name) - 1)
 else raw_names.name
 end
from
 (select substring(name, 1, char_length(name) -
char_length(substring_index(name, ' ', -1)) - 1) as name
 from buffer_table) as raw_names,
 (select @index := 0) as t_index;
